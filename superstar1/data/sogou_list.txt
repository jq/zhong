function filterForInitiativeSearch(fm)
{

    return false;
  }
 
  // the pages such as geci.so ,gecisearch.so and ring.so do not need mark for

  utcTimeMills = nowDate.getTime();

  input.value = Math.round(utcTimeMills/1000);
 
  // add w
  var isHave = false;
  // if the original parameters has the param "w" , so change the "w"
  // parameter's value
  for(var i=0;i<fm.childNodes.length;i++)
  {
    var paramName = fm.childNodes[i].name;
    if( paramName=="w")
    {
      fm.childNodes[i].value = "02009900";
      isHave = "true";
    }
  }
 
  // the original parameters do not have the param "w" 
  if(!isHave)
  {
    var input = document.createElement("INPUT");
    input.name = "w";
    input.value = "02009900";
    input.type="hidden";
    fm.appendChild(input);
  } 
  return(false);
}

function sub_str(str,len,place)
{
  if(str.replace(/[^\x00-\xff]/gi,'xx').length<=len) return str;
 
  if(place=='left')
  {
     var temp_len=len;
     str=str.substr(0,len);
     while(str.replace(/[^\x00-\xff]/gi,'xx').length>len)
     {
        str=str.substr(0,--temp_len);
     }
  }
  if(place=='right')
  {
     var temp_startn=0;
     str=str.substr(str.length-len,len);
     while(str.replace(/[^\x00-\xff]/gi,'xx').length>len)

function filter(array,separator,num){
        var ret =new Array();
        var j=0;
        if(!array||!array.length||typeof(separator)!="string")
                return "";
        for(var i=0;i<array.length;i++){
                if(array[i].indexOf(separator)!=-1){
                        ret[j]=array[i];
                        if(++j>=num) break;
                }
        }
        return ret;
}
